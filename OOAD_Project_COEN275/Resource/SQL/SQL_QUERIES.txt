CREATE TABLE RMOS (RMOSID varchar(50) NOT NULL, PRIMARY KEY(RMOSID));

CREATE TABLE RCMRecycle(RCMNUM int(50) NOT NULL,RCMID VARCHAR(50) NOT NULL,RMOSID varchar(50) NOT NULL, LOCATION VARCHAR (50), STATUS VARCHAR (50), CURRENT_WEIGHT FLOAT(50), TRANSACTION_WEIGHT FLOAT(50), AVAILABLE_CASH FLOAT(50), AVAILABLE_COUPON FLOAT(50), LAST_EMPTIED DATE, PRIMARY KEY(RCMID),FOREIGN KEY (RMOSID) REFERENCES RMOS (RMOSID));

CREATE TABLE RecycleItem(ITEMID INT(50) NOT NULL, ITEMNANE varchar(50) NOT NULL, ITEMTYPE VARCHAR (50), ITEM_WEIGHT FLOAT(50),ITEM_UNIT_WEIGHT FLOAT(50), UNIT_PRICE FLOAT(50));
